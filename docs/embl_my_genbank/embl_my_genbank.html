embl_my_genbank.embl_my_genbank

This module can be used as a library for the command line interface emb_my_gbk, or as an independent python script. It takes an arbitrarily complicated Genbank flat file (we tested on a Geneious Prime-exported file with many added annotations and feature qualifiers) and converts it to a clean, barebones EMBL file. It can optionally convert this EMBL file with additional restrictions to the Immuno-Polymorphism Database flavor of the EMBL format.

usage: embl_my_genbank.py [-h] --gb_path GB_PATH --species SPECIES [--out_fmt OUT_FMT] [--view_intermediate VIEW_INTERMEDIATE]

options:
  -h, --help            show this help message and exit
  --gb_path GB_PATH, -g GB_PATH
                        Genbank file to be converted.
  --species SPECIES, -s SPECIES
                        Scientific name for the species under examination.
  --out_fmt OUT_FMT, -o OUT_FMT
                        Format to convert to. Can convert to EMBL, IPD_EMBL, and FASTA.
  --view_intermediate VIEW_INTERMEDIATE, -v VIEW_INTERMEDIATE

 88def choose_out_type(format_choice: str) -> str:
 89    """
 90    Determines the output file format based on the user's choice.
 91
 92    This function maps a user-provided string indicating the desired
 93    output file format to the corresponding format string used by Biopython
 94    for file writing. It validates the input against a predefined list of
 95    acceptable formats and raises an error for invalid inputs.
 96
 97    Args:
 98        format_choice (str): A string representing the user's choice of output
 99        file format. Expected values correspond to the keys of the `OutType` enum
100        (e.g., 'EMBL', 'IPD_EMBL', 'GENBANK', 'FASTA').
101
102    Returns:
103        str: The string identifier for the output file format as recognized by
104        Biopython (e.g., 'embl', 'genbank', 'fasta').
105
106    Raises:
107        ValueError: If the provided `format_choice` does not correspond to a valid
108        output file format, indicating either a typo or an unsupported format choice.
109
110    Example:
111        >>> choose_out_type('fasta')
112        'fasta'
113        >>> choose_out_type('genbank')
114        'genbank'
115        >>> choose_out_type('invalid')  # This will raise a ValueError
116
117    Note:
118        The function is case-insensitive for `format_choice` but expects it to match
119        one of the predefined format keys exactly.
120    """
121    try:
122        choice_str = OutType.__members__[format_choice.upper()]
123    except KeyError as exc:
124        raise ValueError(f"Invalid file type selected: {format_choice}") from exc
125
126    out_type: str
127    match choice_str:
128        case OutType.EMBL:
129            out_type = "embl"
130        case OutType.IPD_EMBL:
131            out_type = "embl"
132        case OutType.GENBANK:
133            out_type = "genbank"
134        case OutType.FASTA:
135            out_type = "fasta"
136        case _:
137            raise ValueError(f"Invalid output file type selected: {format_choice}")
138    return out_type

141def handle_allele_id_placement(record: SeqRecord) -> SeqRecord:
142    """
143    Corrects the placement of allele identifiers in sequence records exported
144    from Geneious Prime.
145
146    Geneious Prime sometimes places allele names in a location that does not
147    conform to expected standards for certain analyses or database submissions.
148    This function addresses the issue by relocating the allele name from its original
149    position to both the record's ID and accession fields. It ensures that the allele
150    name is consistently represented across the sequence record, facilitating proper
151    identification and reference.
152
153    Args:
154        record (SeqRecord): The sequence record to be corrected. This `SeqRecord` object,
155        as defined by the Biopython library, represents a single sequence and its associated
156        data. The record is expected to have an allele name as part of its naming convention,
157        which will be utilized to update the ID and accession fields.
158
159    Returns:
160        SeqRecord: The modified `SeqRecord` object with the allele name correctly positioned
161        within the ID and accession annotations. This adjustment ensures that the allele name
162        is prominently and correctly displayed in the record, aligning with standard practices
163        for sequence identification.
164
165    Raises:
166        AssertionError: If the allele name is not properly propagated to the accession annotation
167        after the function's execution, an AssertionError is raised. This serves as a check to
168        ensure that the intended modifications have been successfully applied to the sequence record.
169
170    Example:
171        >>> from Bio.Seq import Seq
172        >>> from Bio.SeqRecord import SeqRecord
173        >>> original_record = SeqRecord(Seq("AGTC"), id="geneious_id", name="Allele_A1")
174        >>> corrected_record = handle_allele_id_placement(original_record)
175        >>> corrected_record.id
176        'Allele_A1'
177        >>> corrected_record.annotations["accessions"]
178        ['Allele_A1']
179
180    Note:
181        This function modifies the input `SeqRecord` object in place by adjusting its ID and
182        annotations. It is designed to ensure that allele names are correctly positioned within
183        the sequence record, reflecting best practices for sequence data management and submission.
184    """
185
186    # parse out the working name of the allele
187    allele = record.name
188
189    # parse out the gene name from the allele name
190    # gene = allele.split("_")[1]
191
192    # move the allele name into the id position, replacing a geneious annotation
193    record.id = allele
194
195    # rename the accession to be the allele name
196    record.annotations["accessions"] = [allele]
197
198    # check that the allele name was properly propogated
199    assert (
200        record.annotations["accessions"][0] == record.name
201    ), "Record name could not be properly assigned to accession annotation"
202
203    return record

214
215    Args:

217        object as defined by the Biopython library which represents a single sequence
218        and its associated data, including annotations. species (str): The scientific
219        name of the species to be assigned to the sequence record. This name is used
220        to update the 'organism' field within the record's annotations, ensuring that
221        the record accurately reflects the specified species.
222
223    Returns:
224        SeqRecord: The modified `SeqRecord` object with the 'organism' annotation updated
225        to the specified species' scientific name. This ensures that the sequence record
226        is correctly annotated with the appropriate species information for downstream
227        applications.
228
229    Example:
230        >>> from Bio.Seq import Seq
231        >>> from Bio.SeqRecord import SeqRecord
232        >>> record = SeqRecord(Seq("ACTG"), id="example_id", description="An example sequence.")

234        >>> updated_record.annotations["organism"]
235        'Homo sapiens'

238        The function modifies the input `SeqRecord` object in place by updating its

248def remove_geneious_annotations(record: SeqRecord) -> SeqRecord:
249    """
250    Removes Geneious-specific annotations from a sequence record.
251
252    Geneious software is known to add certain annotations to sequence records that may not
253    be desired in all contexts. This function specifically targets and removes:
254    
255    1. "misc_feature" features that Geneious adds, often related to manual editing history.
256    2. "/note" qualifiers within any feature, which can include comments or metadata added
257    by Geneious.
258
259    The removal of these annotations can help in simplifying the sequence record for downstream
260    analysis or for submission to databases where such annotations are not required or could
261    cause confusion.
262
263    Args:
264        record (SeqRecord): The sequence record from which Geneious-specific annotations are
265            to be removed. The record should be a SeqRecord object as defined by the Biopython
266            library, potentially containing features and qualifiers added by Geneious software.
267
268    Returns:
269        SeqRecord: The same SeqRecord object passed as input, but modified to remove Geneious-
270            specific annotations. Specifically, all "misc_feature" features are removed, and any
271            "note" qualifiers within other features are deleted. This ensures the record is cleaner
272            and more compliant with general sequence record standards.
273
274    Example:
275        >>> from Bio.SeqRecord import SeqRecord
276        >>> from Bio.SeqFeature import SeqFeature, FeatureLocation
277        >>> record = SeqRecord(seq="ACTG", id="example")
278        >>> feature = SeqFeature(FeatureLocation(start=0, end=4), type="misc_feature", qualifiers={"note": "Geneious annotation"})
279        >>> record.features.append(feature)
280        >>> cleaned_record = remove_geneious_annotations(record)
281        >>> len(cleaned_record.features)
282        0
283    
284    Note:
285        The function modifies the input SeqRecord object in place, but also returns it for convenience.
286        This allows for the modified record to be reassigned to the same or a different variable if
287        desired.
288    """
289
290    # get rid of the misc features Geneious adds in for where parts of annotations
291    # were manually deleted
292    record.features = list(
293        filterfalse(lambda f: f.type == "misc_feature", record.features)
294    )
295
296    # delete geneious notes
297    for feature in record.features:
298        # Check if the feature has a 'note' qualifier and remove it
299        if "note" not in feature.qualifiers:
300            continue
301        del feature.qualifiers["note"]
302
303    return record

306def clean_record_features(record: SeqRecord, species: str) -> SeqRecord:
307    """
308        Geneious also exports a number of features and feature qualifiers that are
309        unnecessary for our purposes, namely "standard_name" and "gene". This function does
310        away with those as well, and also adds species and DNA information to the source
311        feature, if available.
312    
313    Args:
314        record (SeqRecord): The sequence record to be cleaned and updated. A SeqRecord object
315            from Biopython that contains features which may include "gene", "CDS", and "source"
316            types among others.
317        species (str): The species name to be used for updating the "organism" qualifier in the
318            "source" feature of the record. This string should represent the scientific name of
319            the organism, e.g., "Homo sapiens".
320
321    Returns:
322        SeqRecord: The modified SeqRecord object with cleaned features according to the
323            specifications. The returned record will have "gene" features removed, "standard_name"
324            qualifiers deleted from "CDS" features where applicable, and the "source" feature
325            updated with the provided species name and molecule type set to "genomic DNA".
326
327    Note:
328        The function directly modifies the input SeqRecord object and also returns this modified
329        object. Therefore, the changes will be reflected in the original SeqRecord passed to the function.
330    """
331
332    # filter out genes
333    record.features = list(filterfalse(lambda f: f.type == "gene", record.features))
334
335    for feature in record.features:
336        # Remove standard_name qualifier from CDS features
337        if feature.type == "CDS" and "standard_name" in feature.qualifiers:
338            del feature.qualifiers["standard_name"]
339
340        # Update source feature with organism and mol_type
341        if feature.type == "source":
342            feature.qualifiers["organism"] = [species]
343            feature.qualifiers["molecule_type"] = ["genomic DNA"]
344
345    return record

348def clean_genbank(record: SeqRecord, species: str) -> SeqRecord:
349    """
350    Cleans a GenBank record by updating allele IDs, assigning species names, and
351    removing Geneious annotations.
352
353    This function performs a series of cleaning and updating steps on a given GenBank
354    sequence record. These steps include correcting the placement of allele IDs, assigning
355    a scientific name to the species annotation, removing specific Geneious annotations,
356    and cleaning various feature annotations added by Geneious software.
357
358    Args:
359        record (SeqRecord): The sequence record to be cleaned and updated.
360        species (str): The scientific name of the species to be assigned to the record.
361
362    Returns:
363        SeqRecord: The cleaned and updated sequence record.
364
365    The function is a wrapper that sequentially applies several specific cleaning and
366    updating operations, making it convenient to perform multiple adjustments through a
367    single function call.
368    """
369
370    # shift around the allele ID into different slots than
371    # Geneious places it by default (i.e., Accession and )
372    record = handle_allele_id_placement(record)
373
374    # assign the user provided scientific name for annotation
375    # onto the record
376    record = assign_species(record, species)
377
378    # clean off geneious annotations
379    record = remove_geneious_annotations(record)
380
381    # clean up the many features appended to each Genbank record,
382    # which BioPython represents as a dictionary
383    record = clean_record_features(record, species)
384
385    return record

388def id_line_replacement(
389    records: List[SeqRecord], int_embl: str, final_out: str, out_type: str
390) -> None:
391    """
392    Replaces the ID line in EMBL-formatted files and writes the modified
393    records to a new file.
394
395    This function first creates an intermediate file from the given sequence
396    records in the specified output format. It then reads this intermediate file,
397    replaces lines starting with "ID" with a predetermined new ID line, and writes
398    the results to a final output file.
399
400    Args:
401        records (List[SeqRecord]): A list of SeqRecord objects to be processed.
402        int_embl (str): The path for the intermediate file to be created.
403        final_out (str): The path for the final output file with modified ID lines.
404        out_type (str): The format of the output file (e.g., 'embl', 'genbank').
405
406    The function is useful for adjusting file formats that require specific header
407    formats not directly supported by the original exporting tool.
408    """
409
410    new_id_line: str = "ID   XXX; XXX; linear; XXX; XXX; XXX;\n"
411
412    # make an intermediate embl file first
413    with open(int_embl, "w", encoding="utf8") as int_handle:
414        for record in records:
415            SeqIO.write(record, int_handle, out_type)
416
417    # Open the input file for reading and the output file for writing
418    with open(int_embl, "r", encoding="utf8") as infile, open(
419        final_out, "w", encoding="utf8"
420    ) as outfile:
421        for line in infile:
422            # Check if the line starts with "ID"
423            if line.startswith("ID"):
424                # Write the replacement text to the output file
425                outfile.write(new_id_line)
426                continue
427            # Write the original line to the output file
428            outfile.write(line)

431def write_output(
432    records: List[SeqRecord],
433    final_out: str,
434    requested_format: str,
435    view_intermediate: bool,
436) -> None:
437    """
438    Writes sequence records to an output file in a specified format, with
439    options for intermediate viewing and format adjustments.
440
441    This function supports writing the sequence records to a file in various
442    formats. It includes options for creating an intermediate file for review
443    and handling special format requirements such as ID line replacements in EMBL
444    files for IPC entries.
445
446    Args:
447        records (List[SeqRecord]): A list of SeqRecord objects to be written to the file.
448        final_out (str): The path for the output file.
449        requested_format (str): The desired format for the output file (e.g., 'EMBL',
450        'IPD_EMBL', 'GENBANK', 'FASTA').
451        view_intermediate (bool): Whether to create and view an intermediate file before
452        final output. Useful for debugging or intermediate analysis.
453
454    The function adapts to the requested format, providing flexibility in output file
455    generation and format-specific handling.
456    """
457
458    out_type = choose_out_type(requested_format)
459
460    if view_intermediate:
461        SeqIO.write(records, "intermediate.gbk", "genbank")
462
463    # check if an IPC intermediate embl is needed
464    if "IPD" in requested_format.upper():
465        int_embl = "intermediate.emb"
466        id_line_replacement(records, int_embl, final_out, out_type)
467        return
468
469    with open(final_out, "w", encoding="utf8") as out_file:
470        for record in records:
471            SeqIO.write(record, out_file, out_type)

474def main() -> None:
475    """
476    Coordinate the flow of data through the embl_my_genbank functions.
477    """
478
479    # parse command line arguments
480    gb_path, species, out_format, view_intermediate = parse_command_line_args()
481
482    assert os.path.isfile(gb_path), "Provided file path does not exist"
483
484    # parse out a name for the output file based on the input file
485    basename = os.path.basename(gb_path).replace(".gb", "")
486    out_path = f"{basename}.emb"
487
488    # revise and clean all records in the Genbank file
489    records = [
490        clean_genbank(record, species) for record in SeqIO.parse(gb_path, "genbank")
491    ]
492
493    # write output in the desired format
494    write_output(records, out_path, out_format, view_intermediate)
495
496
497if __name__ == "__main__":
498    main()

307def clean_record_features(record: SeqRecord, species: str) -> SeqRecord:
308    """
